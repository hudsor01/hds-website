<!-- Pattern -->

```typescript
import { after } from 'next/server'

export async function POST(req: Request) {
  const body = await req.json()

  // Critical path — must finish before response
  const result = await db.insert(contacts).values(body).returning()

  // Fire-and-forget — defer
  after(async () => {
    try {
      await sendAdminNotification(result[0])
      logger.info('contact.admin_notified', { contactId: result[0].id })
    } catch (err) {
      logger.error('contact.admin_notification_failed', { err })
    }
  })

  return Response.json({ ok: true, id: result[0].id })
}
```

Key rules:
- `after()` callbacks run AFTER the response is sent — they cannot modify the response
- Errors inside `after()` MUST be caught and logged; an unhandled rejection in an `after()` callback is a silent failure
- Don't wrap a `db.insert` in `after()` if the response includes the inserted row's id — that's a critical-path operation
