export const PAGE_SIZE = 25

export type Direction = 'after' | 'before'

export type Cursor = {
  direction: Direction
  parts: string[]
}

/**
 * Encode a cursor for a row whose ORDER BY tuple is `parts`.
 * `parts` may contain Date | number | string; each is stringified
 * (Date -> ISO string, number -> String(n)) before joining with `:`.
 * The full payload is then base64url-encoded so colons inside ISO
 * timestamps do not collide with the direction separator.
 *
 * Output shape: `${direction}:${base64urlPayload}`
 * Example: 'after:MjAyNi0wNS0yN1QxMjozNDo1Ni43ODlaOmFiYy11dWlk'
 */
export function encodeCursor(
  direction: Direction,
  parts: (string | number | Date)[]
): string

/**
 * Decode a cursor produced by encodeCursor. Returns null when:
 *  - raw is undefined or empty
 *  - missing the `direction:payload` separator
 *  - direction is not 'after' | 'before'
 *  - base64url decode fails
 *  - payload is empty
 */
export function decodeCursor(raw: string | undefined): Cursor | null

/**
 * Escape SQL LIKE wildcards so a literal `%` in the user query
 * matches a literal `%` in the column. Order matters: escape `\`
 * first so the wildcard escapes below do not double-escape.
 */
export function escapeLikePattern(q: string): string

    ```typescript
    /**
     * Phase 10 admin-list cursor codec + SQL LIKE escape.
     *
     * Server-only: imported by `src/lib/admin/*-queries.ts`. Pure functions,
     * no DB / IO. The cursor format is `${direction}:${base64urlPayload}` where
     * the payload joins the ORDER BY tuple with `:` — colons inside ISO
     * timestamps cannot collide with the direction separator because the
     * payload is base64url-encoded before being concatenated.
     *
     * decodeCursor never throws: malformed input returns null and the caller
     * treats it as "no cursor" (page 1). This keeps a hand-edited URL from
     * crashing the admin shell.
     */
    import 'server-only'

    export const PAGE_SIZE = 25

    export type Direction = 'after' | 'before'

    export type Cursor = {
      direction: Direction
      parts: string[]
    }

    const PART_SEPARATOR = ':'
    const DIRECTION_SEPARATOR = ':'

    function toBase64Url(input: string): string {
      return Buffer.from(input, 'utf8')
        .toString('base64')
        .replace(/\+/g, '-')
        .replace(/\//g, '_')
        .replace(/=+$/, '')
    }

    function fromBase64Url(input: string): string | null {
      // Base64url uses only [A-Za-z0-9_-]; reject anything else outright so
      // a URL-injected value can't slip past Buffer's lenient decoding.
      if (!/^[A-Za-z0-9_-]+$/.test(input)) {
        return null
      }
      const padded = input.replace(/-/g, '+').replace(/_/g, '/')
      const padLen = (4 - (padded.length % 4)) % 4
      try {
        return Buffer.from(padded + '='.repeat(padLen), 'base64').toString('utf8')
      } catch {
        return null
      }
    }

    function stringifyPart(part: string | number | Date): string {
      if (part instanceof Date) {
        return part.toISOString()
      }
      return String(part)
    }

    export function encodeCursor(
      direction: Direction,
      parts: (string | number | Date)[]
    ): string {
      const payload = parts.map(stringifyPart).join(PART_SEPARATOR)
      return `${direction}${DIRECTION_SEPARATOR}${toBase64Url(payload)}`
    }

    export function decodeCursor(raw: string | undefined): Cursor | null {
      if (!raw) {
        return null
      }
      const idx = raw.indexOf(DIRECTION_SEPARATOR)
      if (idx < 0) {
        return null
      }
      const direction = raw.slice(0, idx)
      const payloadEncoded = raw.slice(idx + 1)
      if (direction !== 'after' && direction !== 'before') {
        return null
      }
      if (payloadEncoded.length === 0) {
        return null
      }
      const payload = fromBase64Url(payloadEncoded)
      if (payload === null) {
        return null
      }
      if (payload.length === 0) {
        return null
      }
      return {
        direction,
        parts: payload.split(PART_SEPARATOR)
      }
    }

    export function escapeLikePattern(q: string): string {
      // Order matters: escape backslash FIRST so the wildcard escapes below
      // do not get double-escaped on their own backslash prefix.
      return q
        .replaceAll('\\', '\\\\')
        .replaceAll('%', '\\%')
        .replaceAll('_', '\\_')
    }
    ```
