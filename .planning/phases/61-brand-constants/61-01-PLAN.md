<!-- Hand-rolled OKLCH -> sRGB conversion (Björn Ottosson formulas) — see RESEARCH.md for derivation -->

```typescript
// In scripts/generate-brand-tokens.ts

interface OklchInput { l: number; c: number; h: number }

function oklchToHex({ l, c, h }: OklchInput): string {
  // 1. Cylindrical -> rectangular
  const aLab = c * Math.cos((h * Math.PI) / 180)
  const bLab = c * Math.sin((h * Math.PI) / 180)

  // 2. OKLab -> linear sRGB
  const l_ = (l + 0.3963377774 * aLab + 0.2158037573 * bLab) ** 3
  const m_ = (l - 0.1055613458 * aLab - 0.0638541728 * bLab) ** 3
  const s_ = (l - 0.0894841775 * aLab - 1.2914855480 * bLab) ** 3

  const rLin = +4.0767416621 * l_ - 3.3077115913 * m_ + 0.2309699292 * s_
  const gLin = -1.2684380046 * l_ + 2.6097574011 * m_ - 0.3413193965 * s_
  const bLin = -0.0041960863 * l_ - 0.7034186147 * m_ + 1.7076147010 * s_

  // 3. Gamma correction
  const toSrgb = (x: number) =>
    x <= 0.0031308 ? 12.92 * x : 1.055 * x ** (1 / 2.4) - 0.055

  const r = Math.round(Math.min(1, Math.max(0, toSrgb(rLin))) * 255)
  const g = Math.round(Math.min(1, Math.max(0, toSrgb(gLin))) * 255)
  const b = Math.round(Math.min(1, Math.max(0, toSrgb(bLin))) * 255)

  // 4. Hex
  return `#${[r, g, b].map(v => v.toString(16).padStart(2, '0')).join('')}`
}
```

<!-- Token parsing regex -->

```typescript
const TOKEN_REGEX = /^\s*(--color-[a-z0-9-]+):\s*oklch\(\s*([^)]+?)\s*\)\s*;/gm

function parseOklchValue(raw: string): OklchInput {
  // raw e.g. "0.38 0.12 255" or "38% 0.12 255"
  const parts = raw.trim().split(/\s+/)
  const l = parts[0].endsWith('%') ? parseFloat(parts[0]) / 100 : parseFloat(parts[0])
  const c = parseFloat(parts[1])
  const h = parseFloat(parts[2])
  return { l, c, h }
}
```
