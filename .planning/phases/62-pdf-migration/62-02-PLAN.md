<!-- Pattern -->

```typescript
// Before:
import { Document, Page, StyleSheet, Text, View } from '@react-pdf/renderer'

const styles = StyleSheet.create({
  header: {
    borderBottomColor: '#0891b2'  // cyan-600
  },
  title: {
    color: '#0891b2'  // cyan-600
  }
})

// After:
import { Document, Page, StyleSheet, Text, View } from '@react-pdf/renderer'
import { BRAND } from '@/lib/_generated/brand'

const styles = StyleSheet.create({
  header: {
    borderBottomColor: BRAND.primary
  },
  title: {
    color: BRAND.primary
  }
})
```

Note: `BRAND.primary` is a hex string (e.g., `'#2f4f7a'`) — not an oklch() call. React-PDF's StyleSheet only accepts hex/rgb literals. The hex value comes from `src/lib/_generated/brand.ts` which is mechanically derived from `globals.css` `--color-primary`.
